113  private void UpdateAttachments()
114  {
115  ModifierPlayer modplr = ModifierPlayer.Player(player);
116 
117  // Manual detach
118  var detachEffects = GetModifierEffectsForDelegations(_detachList, modplr, (e) => e.IsBeingDelegated && !_modifierEffects.Contains(e.GetType()));
119  var attachEffects = GetModifierEffectsForDelegations(_attachList, modplr, (e) => !e.IsBeingDelegated);
120  // Automatic delegation lists
121  var orderedDetachList = OrderDelegationList(_detachList, modplr)
122  .Where(x => x.Effect.IsBeingDelegated && !_modifierEffects.Contains(x.Effect.GetType()))
123  .GroupBy(x => x.MethodInfo)
124  .Select(x => x.First())
125  .ToList();
126 
127  var orderedAttachList = OrderDelegationList(_attachList, modplr)
128  .Where(x => !x.Effect.IsBeingDelegated)
129  .GroupBy(x => x.MethodInfo)
130  .Select(x => x.First())
131  .ToList();
132 
133  // Manual detach
134  foreach (var effect in detachEffects.Distinct())
135  {
136  modplr.OnResetEffects -= effect.ResetEffects;
137  effect._DetachDelegations(modplr);
138  effect.IsBeingDelegated = false;
139  }
140 
141  // Manual attach
142  foreach (var effect in attachEffects.Distinct())
143  {
144  modplr.OnResetEffects += effect.ResetEffects;
145  effect.AttachDelegations(modplr);
146  effect.IsBeingDelegated = true;
147  }
148 
149  // Auto delegation detach
150  foreach (var info in orderedDetachList)
151  {
152  var attr = info.MethodInfo.GetCustomAttribute<AutoDelegation>();
153  attr.Detach(modplr, info.MethodInfo, info.Effect);
154  }
155 
156  // Auto delegation attach
157  foreach (var info in orderedAttachList)
158  {
159  var attr = info.MethodInfo.GetCustomAttribute<AutoDelegation>();
160  attr.Attach(modplr, info.MethodInfo, info.Effect);

169  private List<OrderedDelegationEntry> OrderDelegationList(List<AutoDelegationEntry> list, ModifierPlayer modPlayer)
170  {
171  var tempList = new List<OrderedDelegationEntry>();
172  var tempEarlyMethods = new List<OrderedDelegationEntry>();
173  var tempMiddleMethods = new List<OrderedDelegationEntry>();
174  var tempLateMethods = new List<OrderedDelegationEntry>();
175 
176  var effects = GetModifierEffectsForDelegations(list, modPlayer, (e) => true);
177 
178  foreach (var modEffect in effects)
179  {
180  var delegatedMethods = modEffect
181  .GetType()
182  .GetMethods(BindingFlags.NonPublic | BindingFlags.Public | BindingFlags.Instance)
183  .Where(x => x.GetCustomAttributes(typeof(AutoDelegation), false).Length > 0)
184  .ToArray();
185 
186  tempEarlyMethods.AddRange(delegatedMethods.Where(x =>
187  {
188  var attr = x.GetCustomAttribute(typeof(DelegationPrioritizationAttribute));
189  if (attr != null && (attr as DelegationPrioritizationAttribute).DelegationPrioritization == DelegationPrioritization.Early)
190  {
191  return true;
192  }
193 
194  return false;
195  }).Select(x => new OrderedDelegationEntry
196  {
197  MethodInfo = x,
198  Effect = modEffect
199  }));
200 
201  tempMiddleMethods.AddRange(delegatedMethods.Where(x =>
202  {
203  var attr = x.GetCustomAttribute(typeof(DelegationPrioritizationAttribute));
204  return attr == null;
205  }).Select(x => new OrderedDelegationEntry
206  {
207  MethodInfo = x,
208  Effect = modEffect
209  }));
210 
211  tempLateMethods.AddRange(delegatedMethods.Where(x =>
212  {
213  var attr = x.GetCustomAttribute(typeof(DelegationPrioritizationAttribute));
214  if (attr != null && (attr as DelegationPrioritizationAttribute).DelegationPrioritization == DelegationPrioritization.Late)
215  {
216  return true;
217  }
218 
219  return false;
220  }).Select(x => new OrderedDelegationEntry
221  {
222  MethodInfo = x,
223  Effect = modEffect
224  }));
225  }
226 
227  tempEarlyMethods = tempEarlyMethods.OrderByDescending(x =>
228  {
229  var attr = x.MethodInfo.GetCustomAttribute(typeof(DelegationPrioritizationAttribute));
230  return ((DelegationPrioritizationAttribute)attr).DelegationLevel;
231  }).ToList();
232 
233  tempLateMethods = tempLateMethods.OrderByDescending(x =>
234  {
235  var attr = x.MethodInfo.GetCustomAttribute(typeof(DelegationPrioritizationAttribute));
236  return ((DelegationPrioritizationAttribute)attr).DelegationLevel;
237  }).ToList();
238 
239  tempList.AddRange(tempEarlyMethods);
240  tempList.AddRange(tempMiddleMethods);
241  tempList.AddRange(tempLateMethods);
242  return tempList;
