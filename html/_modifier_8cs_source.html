147  {
148  string modname = tag.GetString("ModName");
149  Assembly assembly;
150  if (EMMLoader.Mods.TryGetValue(modname, out assembly))
151  {
152  // If we load a null here, it means a modifier is unloaded
153  Modifier m = null;
154 
155  var saveVersion = tag.ContainsKey("ModifierSaveVersion") ? tag.GetInt("ModifierSaveVersion") : 1;
156 
157  string modifierTypeName = tag.GetString("Type");
158 
159  // adapt by save version
160  if (saveVersion == 1)
161  {
162  // in first save version, modifiers were saved by full assembly namespace
163  //m = (ModifierPool)Activator.CreateInstance(assembly.GetType(tag.GetString("Type")));// we modified saving
164  modifierTypeName = modifierTypeName.Substring(modifierTypeName.LastIndexOf('.') + 1);
165  m = EMMLoader.GetModifier(modname, modifierTypeName);
166  }
167  else if (saveVersion == 2)
168  {
169  // from saveVersion 2 and onwards, they are saved by assembly (mod) and type name
170  m = EMMLoader.GetModifier(modname, modifierTypeName);
171  }
172 
173  if (m != null)
174  {
175  // saveVersion 1, no longer needed. Type and Mod is already created by new instance
176  //m.Type = tag.Get<uint>("ModifierType");
177  //m.Mod = ModLoader.GetMod(modname);
178  var p = ModifierProperties._Load(item, tag.GetCompound("ModifierProperties"));
179  m.Properties = m.GetModifierProperties(item);
180  m.Properties.Magnitude = p.Magnitude;
181  m.Properties.Power = p.Power;
182  m.Load(item, tag);
183  return m;
184  }
185 
186  return null;
187  }

189  throw new Exception($"Modifier load error for {modname}");
