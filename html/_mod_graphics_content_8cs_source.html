89  //yeah, it works for every pixel individually

91  //in terraria, A channel signals additive<---> opaque
92  //a pixel with 255 alpha draws fully opaque, a pixel with 0 alpha draws fully additive
93  //so if you do say, 255,0,0,0 color, it would draw a fully additive red
94  //which can be awesome...except when your file saves the alpha channel of supposedly transparent pixels wrong
95  public static Texture2D MultiplyColorsByAlpha(Texture2D texture)
96  {
97  Color[] data = new Color[texture.Width * texture.Height];
98  texture.GetData(data);
99  for (int i = 0; i < data.Length; i++)
100  {
101  Vector4 we = data[i].ToVector4();
102  data[i] = new Color(we.X * we.W, we.Y * we.W, we.Z * we.W, we.W);
103  }
104  texture.SetData(data);
105  return texture;
106  }
